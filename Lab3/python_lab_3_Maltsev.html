Завдання № 1

Написати програму, яка обчислює значення функції Гауса: \[f(x) = \frac{1}{\sigma \sqrt{2 \pi}} e^{-\frac{(x-\mu)^2}{2\sigma^2}}\]


import math

def gaus(a1, b1, d1):
    R1 = 1 / (d1 * math.sqrt(2 * math.pi))
    R2 = math.exp(-((a1 - b1) ** 2) / (2 * d1 ** 2))
    return R1 * R2


a = 0
b = 0
d = 1

result = gaus(a, b, d)
print(result)
